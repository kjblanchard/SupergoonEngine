 30 static void createAnimatedTiles(Tilemap *map, Tileset *tileset) {
 31     LuaGetTable(_luaState, "tiles");
 32     tileset->NumAnimatedTiles = LuaGetTableLength(_luaState);
 33     tileset->AnimatedTiles =
 34         calloc(tileset->NumAnimatedTiles, sizeof(AnimatedTile));
 35     for (size_t j = 0; j < tileset->NumAnimatedTiles; j++) {
 36         LuaPushTableObjectToStacki(_luaState,
 37                                    j);  // Actual animated tile is on there now
 38         AnimatedTile *animatedTile = &tileset->AnimatedTiles[j];
 39         animatedTile->GID = LuaGetInt(_luaState, "id") +
 40                             tileset->FirstGid;  // GID is the local ID plus the first
 41                                                 // gid, since it's global
 42         LuaGetTable(_luaState, "animation");
 43         animatedTile->NumFrames = LuaGetTableLength(_luaState);
 44         animatedTile->TileFrames =
 45             calloc(animatedTile->NumFrames, sizeof(TileAnimationFrame));
 46         for (size_t k = 0; k < animatedTile->NumFrames; k++) {
 47             LuaPushTableObjectToStacki(
 48                 _luaState, k);  // Actual animated tile table is on there now
 49             TileAnimationFrame *frameData = &animatedTile->TileFrames[k];
 50             frameData->MsTime = LuaGetInt(_luaState, "duration");
 51             frameData->Id = LuaGetInt(_luaState, "tileid") +
 52                             tileset->FirstGid;  // GID is the local ID plus the first
 53                                                 // gid, since it's global
 54             GetRectForGid(frameData->Id, tileset, &frameData->SrcRect);
 55             LuaPopStack(_luaState, 1);  // remove animated tile frame table from stack
 56         }
 57         LuaPopStack(_luaState, 1);  // remove tileAnimation table from stack
 58         LuaPopStack(_luaState, 1);  // remove animated tile table from stack
 59     }
 60     LuaPopStack(_luaState, 1);  // remove tiles table from stack
 61 }

 63 static void createTilesets(Tilemap *map) {
 64     LuaGetTable(_luaState, "tilesets");
 65     map->NumTilesets = LuaGetTableLength(_luaState);
 66     map->Tilesets = calloc(map->NumTilesets, sizeof(Tileset));
 67     for (int i = 0; i < map->NumTilesets; i++) {
 68         Tileset *tileset = &map->Tilesets[i];
 69         LuaPushTableObjectToStacki(_luaState, i);
 70         tileset->Name = LuaAllocateString(_luaState, "name");
 71         tileset->FirstGid = LuaGetInt(_luaState, "firstgid");
 72         tileset->TileWidth = LuaGetInt(_luaState, "tilewidth");
 73         tileset->TileHeight = LuaGetInt(_luaState, "tileheight");
 74         tileset->Image = LuaAllocateString(_luaState, "image");
 75         tileset->ImageWidth = LuaGetInt(_luaState, "imagewidth");
 76         tileset->ImageHeight = LuaGetInt(_luaState, "imageheight");
 77         createAnimatedTiles(map, tileset);
 78         LuaPopStack(_luaState, 1);  // remove tileset table from stack
 79     }
 80     LuaPopStack(_luaState, 1);  // remove tilesets table from stack
 81 }

 83 static void createLayers(Tilemap *map) {
 84     LuaGetTable(_luaState, "layers");
 85     map->NumLayers = LuaGetTableLength(_luaState);
 86     for (int i = 0; i < map->NumLayers; i++) {
 87         LuaPushTableObjectToStacki(_luaState, i);
 88         const char *layerType = LuaGetString(_luaState, "type");
 89         if (strcmp(layerType, "objectgroup") == 0) {
 90             handleTiledObjectGroup(map);
 91         } else if (strcmp(layerType, "group") == 0) {
 92             handleTiledLayerGroup(map);
 93         }
 94         LuaPopStack(_luaState, 1);  // remove layer object table from stack
 95     }
 96     LuaPopStack(_luaState, 1);  // layers table
 97 }

 99 static Tilemap *parseTiledTilemap(const char *tiledFilename) {
100     char *name;
101     asprintf(&name, "assets/tiled/%s.lua", tiledFilename);
102     LuaPushTableFromFile(_luaState, name);
103     SDL_free(name);
104     Tilemap *map = calloc(1, sizeof(*map));
105     map->BaseFilename = strdup(tiledFilename);
106     map->Width = LuaGetInt(_luaState, "width");
107     map->Height = LuaGetInt(_luaState, "height");
108     map->TileWidth = LuaGetInt(_luaState, "tilewidth");
109     map->TileHeight = LuaGetInt(_luaState, "tileheight");
110     createTilesets(map);
111     createLayers(map);
112     LuaPopStack(_luaState, 1);  // tiled map table
113     LuaClearStack(_luaState);   // for good measure :)
114     return map;
115 }

145 static void loadTilesetTextures(Tilemap *map) {
146     for (size_t i = 0; i < (size_t)map->NumTilesets; i++) {
147         if (map->Tilesets[i].TilesetTexture != NULL) {
148             continue;
149         }
150         int nameLen = strlen(map->Tilesets[i].Image);
151         char *imageName = map->Tilesets[i].Image;
152         // remove bmp extension if needed
153         if (nameLen >= 4 && strcmp(imageName + nameLen - 4, ".bmp") == 0) {
154             imageName[nameLen - 4] = '\0';
155         }
156         map->Tilesets[i].TilesetTexture = TextureCreate();
157         TextureLoadFromBmp(map->Tilesets[i].TilesetTexture, map->Tilesets[i].Image);
158     }
159 }

171 static void handleTiledObjectEntities(Tilemap *map) {
172     LuaGetTable(_luaState, "objects");
173     map->NumObjects = LuaGetTableLength(_luaState);
174     map->Objects = calloc(map->NumObjects, sizeof(TiledObject));
175     for (size_t i = 0; i < (size_t)map->NumObjects; i++) {
176         TiledObject *object = &map->Objects[i];
177         LuaPushTableObjectToStacki(_luaState, i);
178         object->Id = LuaGetInt(_luaState, "id");
179         object->ObjectType = atoi(LuaGetString(_luaState, "type"));
180         object->X = LuaGetFloat(_luaState, "x");
181         object->Y = LuaGetFloat(_luaState, "y");
182         object->Width = LuaGetFloat(_luaState, "width");
183         object->Height = LuaGetFloat(_luaState, "height");
184         LuaGetTable(_luaState, "properties");
185         object->NumProperties = LuaGetTableLengthMap(_luaState);
186         if (object->NumProperties == 0) {
187             LuaPopStack(_luaState, 2);
188             continue;
189         }
190         object->Properties = calloc(object->NumProperties, sizeof(TiledProperty));
191         LuaStartTableKeyValueIteration(_luaState);
192         for (size_t j = 0; j < (size_t)object->NumProperties; j++) {
193             if (!LuaNextTableKeyValueIterate(_luaState)) {
194                 break;
195             }
196             TiledProperty *property = &object->Properties[j];
197             property->Name = LuaAllocateStringStack(_luaState, -2);
198             property->PropertyType = getPropertyTypeForStack();
199             if (property->PropertyType == TiledPropertyTypeInt) {
200                 property->Data.IntData = LuaGetIntFromStack(_luaState);
201             } else if (property->PropertyType == TiledPropertyTypeFloat) {
202                 property->Data.FloatData = LuaGetFloatFromStack(_luaState);
203             } else {
204                 property->Data.StringData = LuaAllocateStringStack(_luaState, -1);
205             }
206             LuaPopStack(_luaState, 1);
207         }
208         LuaEndTableKeyValueIteration(_luaState);
209         LuaPopStack(_luaState, 2);
210     }
211     LuaPopStack(_luaState, 1);
212 }
213 static void handleTiledSolidObjects(Tilemap *map) {
214     LuaGetTable(_luaState, "objects");
215     map->NumSolids = LuaGetTableLength(_luaState);
216     // We are also adding solids for the walls of the map
217     map->Solids = calloc(map->NumSolids + NUM_WALLS, sizeof(RectangleF));
218     for (size_t i = 0; i < (size_t)map->NumSolids; i++) {
219         RectangleF *rect = &map->Solids[i];
220         LuaPushTableObjectToStacki(_luaState, i);
221         rect->x = LuaGetFloat(_luaState, "x");
222         rect->y = LuaGetFloat(_luaState, "y");
223         rect->w = LuaGetFloat(_luaState, "width");
224         rect->h = LuaGetFloat(_luaState, "height");
225         LuaPopStack(_luaState, 1);
226     }
227     LuaPopStack(_luaState, 1);
228     const int boxSize = 16;
229     int mapSizeX = map->Width * map->TileWidth;
230     int mapSizeY = map->Height * map->TileHeight;
231     RectangleF walls[] = {
232         {0, -boxSize, mapSizeX, boxSize},
233         {mapSizeX, 0, boxSize, mapSizeY},
234         {0, mapSizeY, mapSizeX, boxSize},
235         {-boxSize, 0, boxSize, mapSizeY},
236     };
237     memcpy(&map->Solids[map->NumSolids], walls, sizeof(walls));
238     map->NumSolids += NUM_WALLS;
239 }

250 static Tileset *GetTilesetForGID(int gid, Tilemap *map) {
251     int highestGID = 0;
252     Tileset *highestGIDTileset = NULL;
253     for (size_t i = 0; i < (size_t)map->NumTilesets; i++) {
254         if (gid >= map->Tilesets[i].FirstGid &&
255             map->Tilesets[i].FirstGid >= highestGID) {
256             highestGID = map->Tilesets[i].FirstGid;
257             highestGIDTileset = &map->Tilesets[i];
258         }
259     }
260     return highestGIDTileset;
261 }

263 static void GetRectForGid(int gid, Tileset *tileset, RectangleF *rect) {
264     int tilemapGid = gid - tileset->FirstGid;
265     int x = (tilemapGid % (tileset->ImageWidth / tileset->TileWidth)) *
266             tileset->TileWidth;
267     int y = (tilemapGid / (tileset->ImageWidth / tileset->TileWidth)) *
268             tileset->TileHeight;
269     rect->x = x;
270     rect->y = y;
271     rect->w = tileset->TileWidth;
272     rect->h = tileset->TileHeight;
273 }

275 /**
276  * @brief Get the Animated Tile For Gid object
277  * @param gid gid to search for
278  * @param tileset tileset to search in for animated tiles gid
279  * @return AnimatedTile* the tile, or null if it isn't found
280  */
281 static AnimatedTile *getAnimatedTileForGid(unsigned int gid, Tileset *tileset) {
282     for (size_t i = 0; i < tileset->NumAnimatedTiles; i++) {
283         if (gid == tileset->AnimatedTiles[i].GID) {
284             return &tileset->AnimatedTiles[i];
285         }
286     }
287     return NULL;
288 }

290 // Animated tiles do not get drawn to the map, instead they get added to the
291 // animated tile draw rectangles array.
292 static void handleAnimatedTile(Tilemap *map, Tileset *srcTileset,
293                                RectangleF *dstRect,
294                                AnimatedTile *animatedTile) {
295     ++animatedTile->NumDrawRectangles;
296     RectangleF *newSpace =
297         realloc(animatedTile->DrawRectangles,
298                 sizeof(RectangleF) * animatedTile->NumDrawRectangles);
299     if (!newSpace) {
300         sgLogError("Could not realloc, what in the world probably broken?");
301         return;
302     }
303     animatedTile->DrawRectangles = newSpace;
304     RectangleF *dstRectPtr =
305         &animatedTile->DrawRectangles[animatedTile->NumDrawRectangles - 1];
306     dstRectPtr->x = dstRect->x;
307     dstRectPtr->y = dstRect->y;
308     dstRectPtr->w = dstRect->w;
309     dstRectPtr->h = dstRect->h;
310 }

312 // Called every time we load a map, as the bg1 and bg2 textures must be redrawn.
313 static void createBackgroundsFromTilemap(Tilemap *map) {
314     int mapWidth = map->Width * map->TileWidth;
315     int mapHeight = map->Height * map->TileHeight;
316     if (_bg1Texture) {
317         TextureDestroy(_bg1Texture);
318     }
319     //_bg1Texture = TextureCreate();
320     _bg1Texture = TextureCreateRenderTarget(mapWidth, mapHeight);
321     SetRenderTarget(_bg1Texture);
322     loadTilesetTextures(map);
323     LayerGroup *bg1LayerGroup = &map->LayerGroups[0];
324     RectangleF dstRect = {0, 0, map->TileWidth, map->TileHeight};
325     RectangleF srcRect = {0, 0, 0, 0};
326     for (size_t i = 0; i < (size_t)bg1LayerGroup->NumLayers; i++) {
327         TileLayer *layer = &bg1LayerGroup->Layers[i];
328         for (int y = 0; y < layer->Height; ++y) {
329             for (int x = 0; x < layer->Width; ++x) {
330                 int index = (y * layer->Width) + x;
331                 int tileGid = layer->Data[index];
332                 if (tileGid == 0)
333                     continue;
334                 Tileset *srcTileset = GetTilesetForGID(tileGid, map);
335                 dstRect.x = x * map->TileWidth;
336                 dstRect.y = y * map->TileHeight;
337                 AnimatedTile *animatedTile = getAnimatedTileForGid(tileGid, srcTileset);
338                 if (animatedTile) {
339                     handleAnimatedTile(map, srcTileset, &dstRect, animatedTile);
340                     continue;
341                 }
342                 GetRectForGid(tileGid, srcTileset, &srcRect);
343                 Texture *srcTexture = srcTileset->TilesetTexture;
344                 DrawTextureToTexture(_bg1Texture, srcTexture, GetDefaultShader(), &dstRect, &srcRect, 1.0f);
345             }
346         }
347     }
348     SetRenderTarget(NULL);
349 }

351 static void freeTiledTilemap(Tilemap *map) {
352     for (size_t i = 0; i < 2; i++) {
353         for (size_t j = 0; j < (size_t)map->LayerGroups[i].NumLayers; j++) {
354             SDL_free(map->LayerGroups[i].Layers[j].Data);
355         }
356         SDL_free(map->LayerGroups[i].Name);
357         SDL_free(map->LayerGroups[i].Layers);
358     }
359     if (map->NumObjects > 0) {
360         for (size_t i = 0; i < (size_t)map->NumObjects; i++) {
361             for (size_t j = 0; j < map->Objects[i].NumProperties; j++) {
362                 SDL_free(map->Objects[i].Properties[j].Name);
363                 if (map->Objects[i].Properties[j].PropertyType ==
364                     TiledPropertyTypeString) {
365                     SDL_free(map->Objects[i].Properties[j].Data.StringData);
366                 }
367             }
368             SDL_free(map->Objects[i].Name);
369             SDL_free(map->Objects[i].Properties);
370         }
371         SDL_free(map->Objects);
372     }
373     // each tileset has a list of animated tiles, and also each of those has a
374     // list of frames
375     for (size_t i = 0; i < map->NumTilesets; i++) {
376         Tileset *tileset = &map->Tilesets[i];
377         for (size_t j = 0; j < tileset->NumAnimatedTiles; j++) {
378             AnimatedTile *animatedTile = &tileset->AnimatedTiles[j];
379             SDL_free(animatedTile->TileFrames);
380             SDL_free(animatedTile->DrawRectangles);
381         }
382         SDL_free(tileset->Name);
383         SDL_free(tileset->Image);
384         SDL_free(tileset->AnimatedTiles);
385         TextureDestroy(tileset->TilesetTexture);
386     }
387     if (map->Solids) {
388         SDL_free(map->Solids);
389     }
390     SDL_free(map->Tilesets);
391     SDL_free(map->BaseFilename);
392     SDL_free(map);
393     map = NULL;
394 }

396 static void drawAnimatedTiles(void) {
397     // Update the animated tiles
398     for (size_t i = 0; i < _currentMap->NumTilesets; i++) {
399         Tileset *tileset = &_currentMap->Tilesets[i];
400         for (size_t j = 0; j < tileset->NumAnimatedTiles; j++) {
401             AnimatedTile *animatedTile = &tileset->AnimatedTiles[j];
402             animatedTile->CurrentMSOnFrame += DeltaTimeMilliseconds;
403             while (true) {
404                 TileAnimationFrame *currentFrame =
405                     &animatedTile->TileFrames[animatedTile->CurrentFrame];
406                 if (animatedTile->CurrentMSOnFrame >= currentFrame->MsTime) {
407                     animatedTile->CurrentMSOnFrame -= currentFrame->MsTime;
408                     animatedTile->CurrentFrame =
409                         animatedTile->CurrentFrame + 1 >= animatedTile->NumFrames
410                             ? 0
411                             : animatedTile->CurrentFrame + 1;
412                 } else {
413                     break;
414                 }
415             }
416         }
417     }
418     // Draw the animated tiles
419     for (size_t i = 0; i < _currentMap->NumTilesets; i++) {
420         Tileset *tileset = &_currentMap->Tilesets[i];
421         for (size_t j = 0; j < tileset->NumAnimatedTiles; j++) {
422             AnimatedTile *animatedTile = &tileset->AnimatedTiles[j];
423             for (size_t k = 0; k < animatedTile->NumDrawRectangles; k++) {
424                 RectangleF dst = animatedTile->DrawRectangles[k];
425                 DrawTexture(tileset->TilesetTexture, GetDefaultShader(), &dst, &animatedTile->TileFrames[animatedTile->CurrentFrame].SrcRect, true, 1.0f, false);
426                 // DrawTexture(
427                 //     tileset->TilesetTexture, &dst,
428                 //     &animatedTile->TileFrames[animatedTile->CurrentFrame].SrcRect);
429             }
430         }
431     }
432 }

450 static Tilemap *checkCache(const char *mapName) {
451     if (!_currentMap) {
452         return NULL;
453     }
454 
455     Tilemap *returnMap = NULL;
456     int foundIndex = -1;
457     unsigned int cacheSize = 0;
458 
459     // First pass: look for cache hit
460     for (size_t i = 0; i < MAX_PREVIOUS_MAPS_CACHE; i++) {
461         if (_previousMaps[i] == NULL)
462             break;
463 
464         if (strcmp(mapName, _previousMaps[i]->BaseFilename) == 0) {
465             returnMap = _previousMaps[i];
466             foundIndex = i;
467             break;
468         }
469     }
470 
471     // Second pass: determine actual cache size
472     for (cacheSize = 0;
473          cacheSize < MAX_PREVIOUS_MAPS_CACHE && _previousMaps[cacheSize] != NULL;
474          ++cacheSize);
475 
476     // If cache hit, remove it from cache (shift others left)
477     if (foundIndex >= 0) {
478         for (size_t i = foundIndex; i < cacheSize - 1; i++) {
479             _previousMaps[i] = _previousMaps[i + 1];
480         }
481         _previousMaps[cacheSize - 1] = NULL;
482         // Caller is expected to assign this to _currentMap
483         return returnMap;
484     }
485 
486     // Cache current map if it is different from the map being loaded
487     if (_currentMap && strcmp(mapName, _currentMap->BaseFilename) != 0) {
488         // If cache is full, free the last entry
489         if (cacheSize == MAX_PREVIOUS_MAPS_CACHE) {
490             Tilemap *map = _previousMaps[cacheSize - 1];
491             if (map) {
492                 freeTiledTilemap(map);
493             }
494             cacheSize--;  // Reduce cacheSize to make room
495         }
496 
497         // Shift existing entries down to make space at index 0
498         for (size_t i = cacheSize; i > 0; i--) {
499             _previousMaps[i] = _previousMaps[i - 1];
500         }
501         _previousMaps[0] = _currentMap;
502     } else {
503         // If current map is already the one being loaded, just return it
504         return _currentMap;
505     }
506 
507     return returnMap;
508 }

510 void LoadMap(const char *mapName) {
511     Tilemap *nextMap = checkCache(mapName);
512     if (_currentMap && nextMap == _currentMap) {
513         return;
514     }
515     if (!nextMap) {
516         nextMap = parseTiledTilemap(mapName);
517     }
518     _currentMap = nextMap;
519     createBackgroundsFromTilemap(_currentMap);
520     SetCameraBounds(_currentMap->Width * _currentMap->TileWidth, _currentMap->Height * _currentMap->TileHeight);
521 }

523 void ShutdownMapSystem(void) {
524     if (_bg1Texture) {
525         TextureDestroy(_bg1Texture);
526     }
527     if (_bg2Texture) {
528         TextureDestroy(_bg2Texture);
529     }
530     if (_currentMap) {
531         freeTiledTilemap(_currentMap);
532         _currentMap = NULL;
533     }
534     for (size_t i = 0; i < MAX_PREVIOUS_MAPS_CACHE; i++) {
535         if (_previousMaps[i] == NULL) {
536             break;
537         }
538         freeTiledTilemap(_previousMaps[i]);
539         _previousMaps[i] = NULL;
540     }
